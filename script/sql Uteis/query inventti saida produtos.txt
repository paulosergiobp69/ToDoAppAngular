// consulta das notas fiscais eletronicas de produtos
// as notas rejeitadas tem o campo IND_STATUS_NFE diferente de 6 

SELECT
  *
FROM Nfe.INTERF_NFE
WHERE 1 = 1
AND NUM_DOCTO_FISCAL = '2356'
AND NUM_SERIE_DOCTO_FISCAL = '102'
